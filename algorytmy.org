**** przykład Python
#+begin_src python
def InsertionSort(A):
    n = len(A)
    for j in range (1, n):
        pom = A[j]
        i = j - 1
        while i >= 0 and A[i] > pom:
            A[i+1] = A[i]
            i = i - 1
        A[i+1] = pom
